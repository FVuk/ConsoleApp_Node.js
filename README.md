# ConsoleApp_Node.js

# Get Contact List

node index.js --action list
![Wynik w konsoli](/assets/action%20list.png)

# Get Contact By Id
node index.js --action get --id 05olLMgyVQdWRwgKfg5J6

node index.js --action remove --id qdggE76Jtbfd9eWJHrssH
